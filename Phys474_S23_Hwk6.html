import numpy as np
import matplotlib.pyplot as plt
import scipy.stats as st


def gen_experiment(L):
    T = np.sqrt(L / g)
    return np.random.normal(T, sig_T)

def get_chi2(thry, data):
    return np.sum(((data - thry) / sig_T) ** 2)


lengths = [.1, .25, .5, .75, 1.0, 1.25, 1.5]      # [m]
g = 9.8                                           # [m/s^2]
sig_T = 0.01                                      # [s]

Ls = np.array(lengths)
Ts = gen_experiment(Ls)
T_thry = np.sqrt(Ls / g)

L_thry_plot = np.linspace(Ls.min(), Ls.max(), 100)
T_thry_plot = np.sqrt(L_thry_plot / g)

dof = len(Ls)
chi2 = get_chi2(T_thry, Ts)
prob = st.chi2.sf(chi2, dof)

plt.errorbar(Ls, Ts, yerr=sig_T, fmt='o', label='Experiment')
plt.plot(L_thry_plot, T_thry_plot, label='Theory')
text = r"""$\chi^2={:.3f}$
DOF$={}$
Prob$={:.3f}$""".format(chi2, dof, prob)
plt.text(.8, .02, text, ha='left', va='bottom', transform=plt.gca().transAxes,
         bbox=dict(boxstyle='round', facecolor='wheat', alpha=0.5))

plt.title('Experimental Data for Pendulum Period vs. Length')
plt.xlabel('Length [m]')
plt.ylabel('Period [s]')

plt.grid(linestyle='--')
plt.legend(loc='upper left')
plt.show()

Problem 2: (5 pts)

You are designing an experiment that will measure the characteristic lifetime of a state (e.g. an excited atomic state emitting photons, or nuclear decay emitting charged particle). The number of decays you expect versus time is given by the simple exponential decay function

$N(t) = \frac{N_0}{\tau}e^{-t/\tau}$

where $\tau$ is the characteristic lifetime you would be trying to measure and $N_0$ is the number of initial exctited states. Note that integral of $\int_0^\infty N(t) \, dt = N_0$

To test the sensitivity of your possible measurement of the lifetime $\tau$ to the expected MEAN number of excited states $<N_0>$ you would write a monte carlo simulating $N(t)$ for various actual $N_0$, after Poisson fluctuations. Then you could fit that simulated $N(t)$ for the lifetime $\tau$ over many simulated datasets to estimate your sensitivity for $\tau$ versus $<N_0>$. This would tell you what $<N_0>$ you would need to have in this hypothetical experiment you will design.

N0_mean = 4000
tau = 1           #  [s]
n_bins = 40
trials = 100000

N0 = np.random.poisson(N0_mean)
ts = np.random.exponential(tau, size=N0)
err = np.sqrt(n_bins)

hist, bins, _ = plt.hist(ts, bins=n_bins, label='$N_0={}$'.format(N0))
bins = bins[:-1] + (bins[1] - bins[0]) / 2  # centering bins

N_thry = 1 / tau * np.exp(-bins / tau)
N_thry *= N0_mean / np.sum(N_thry)        # normalizing

chi2 = np.sum(((hist - N_thry) / err) ** 2)
dof = len(hist)
prob = st.chi2.sf(chi2, dof)

plt.plot(bins, N_thry, label='$<N_0>={}$'.format(N0_mean))
plt.errorbar(bins, hist, yerr=err, linestyle='none', color='red', label=r'$N_0$ error')
text = r"""$\chi^2={:.3f}$
DOF$={}$
Prob$={:.3f}$""".format(chi2, dof, prob)
plt.text(.8, .2, text, ha='left', va='bottom', transform=plt.gca().transAxes,
         bbox=dict(boxstyle='round', facecolor='wheat', alpha=0.5))
